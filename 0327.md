#### * 팔레트 생성 
- 조건별 막대의 색 다르게 표현 

```r
v1 <- c(99,81,40,60,89,20)
barplot(v1)
#80이상 빨강,50~80 노랑, 50미만 녹색을 갖는 색 벡터 생성
v_color<-c()
for (i in v1) {
  if (i<50) {
    v_color<-c(v_color,'green')
  } else if (i<80) {
    v_color<-c(v_color,'yellow')
  } else {
    v_color<-c(v_color,'red')
  }
}
barplot(v1,col=v_color,
        names.arg = c('월','화','수','목','금','주말'))
```
